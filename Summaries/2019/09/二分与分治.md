## [Perfect Team](https://codeforces.com/problemset/problem/1221/C)
* 二分即可
```cpp
#include <iostream>
using namespace std;
int c, m, x;
bool Check( int ans, int c, int m, int x )
{
    c -= ans, m -= ans;
    if( c < 0 || m < 0 ) return false;
    int k = m + c + x;
    if( k >= ans ) return true;
    return false;
}
int main()
{
    int q; cin >> q;
    while ( q -- )
    {
        cin >> c >> m >> x;
        int l = 0, r = 1E8 + 5;
        while( l < r )
        {
            int mid = ( l + r + 1 ) >> 1;
            if( Check(mid, c, m, x ) ) l = mid;
            else r = mid - 1;
        }
        cout << l << endl;
    }
}
